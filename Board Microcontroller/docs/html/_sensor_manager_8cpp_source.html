24 // Initialize I2C buses if not already initialized

27 errorHandler->logError(ERROR, "Failed to initialize I2C0");

35 errorHandler->logError(ERROR, "Failed to initialize I2C1");
36 } else {
37 errorHandler->logError(INFO, "Initialized I2C1 bus");
38 }
39 }
40
41 // Initialize SPI if available
42 if (spiManager && !spiManager->isInitialized()) {
43 if (!spiManager->begin()) {
44 errorHandler->logError(ERROR, "Failed to initialize SPI");
45 } else {
46 errorHandler->logError(INFO, "Initialized SPI bus");
47 }
48 }
49
50 // Scan I2C buses for devices
51 std::vector<int> foundAddressesI2C0;
52 std::vector<int> foundAddressesI2C1;
53
54 errorHandler->logError(INFO, "Scanning I2C0 bus for devices...");
55 bool i2c0HasDevices = i2cManager->scanBus(I2CPort::I2C0, foundAddressesI2C0);
56
57 errorHandler->logError(INFO, "Scanning I2C1 bus for devices...");
58 bool i2c1HasDevices = i2cManager->scanBus(I2CPort::I2C1, foundAddressesI2C1);
59
60 if (!i2c0HasDevices && !i2c1HasDevices) {
61 errorHandler->logError(WARNING, "No I2C devices found on any bus - check wiring if using I2C sensors!");
62 }
63
64 // Get sensor configurations
65 auto sensorConfigs = configManager->getSensorConfigs();
66 bool atLeastOneInitialized = false;
67
68 // Clear any existing sensors and update info
69 auto existingSensors = registry.clear();
70 for (auto sensor : existingSensors) {
71 delete sensor;
72 }
73
74 // Create and initialize sensors
75 for (const auto& config : sensorConfigs) {
76 // Test communication before initializing sensor
77 if (config.isSPI) {
78 if (!spiManager) {
79 errorHandler->logError(ERROR, "SPI manager not available for sensor: " + config.name);
80 continue;
81 }
82 testSPICommunication(config.address);
83 } else {

85 }
86
87 // Create the sensor

89 if (!sensor) {
90 errorHandler->logError(ERROR, "Failed to create sensor: " + config.name);
91 continue;
92 }
93
94 // Initialize the sensor
95 if (!sensor->initialize()) {
96 errorHandler->logError(ERROR, "Failed to initialize sensor: " + config.name);
97 delete sensor;
98 continue;
99 }
100
101 // Register the sensor
102 registry.registerSensor(sensor);
103
104 errorHandler->logError(INFO, "Sensor added to system: " + config.name + " with polling rate: " +
105 String(config.pollingRate) + "ms");
106 atLeastOneInitialized = true;
107 }
108
109 if (!atLeastOneInitialized) {
110 errorHandler->logError(ERROR, "No sensors were initialized");
111 return false;
112 }
113
114 return true;
115}

230void SensorManager::compareConfigurations(
231 const std::vector<SensorConfig>& oldConfigs,
232 const std::vector<SensorConfig>& newConfigs,
233 std::vector<SensorConfig>& toAdd,
234 std::vector<String>& toRemove
235) {
236 // Clear output vectors
237 toAdd.clear();
238 toRemove.clear();
239
240 // Find sensors to remove (in old but not in new)
241 for (const auto& oldConfig : oldConfigs) {
242 bool found = false;
243 for (const auto& newConfig : newConfigs) {
244 if (oldConfig.name == newConfig.name) {
245 found = true;
246 break;
247 }
248 }
249
250 if (!found) {
251 toRemove.push_back(oldConfig.name);
252 }
253 }
254
255 // Find sensors to add (in new but not in old)
256 for (const auto& newConfig : newConfigs) {
257 bool found = false;
258 for (const auto& oldConfig : oldConfigs) {
259 if (newConfig.name == oldConfig.name) {
260 // Check if configuration changed
261 if (newConfig.type != oldConfig.type ||
262 newConfig.address != oldConfig.address ||
263 newConfig.isSPI != oldConfig.isSPI ||
264 (!newConfig.isSPI && newConfig.i2cPort != oldConfig.i2cPort)) {
265 // Configuration changed, remove and re-add
266 toRemove.push_back(newConfig.name);
267 toAdd.push_back(newConfig);
268 }
269 found = true;
270 break;
271 }
272 }
273
274 if (!found) {
275 toAdd.push_back(newConfig);
276 }
277 }

416bool SensorManager::reconnectSensor(const String& sensorName) {
417 ISensor* sensor = findSensor(sensorName);
418 if (!sensor) {
419 if (errorHandler) {
420 errorHandler->logError(WARNING, "Cannot reconnect - sensor not found: " + sensorName);
421 }
422 return false;
423 }
424
425 if (sensor->isConnected()) {
426 // Sensor is already connected
427 return true;
428 }
429
430 // Add timeout for reconnection attempt
431 const unsigned long timeout = 500; // 500ms timeout
432 unsigned long startTime = millis();
433
434 if (errorHandler) {
435 errorHandler->logError(INFO, "Attempting to reconnect sensor: " + sensorName);
436 }
437
438 // Check if we've exceeded the timeout at every step
439 if (millis() - startTime >= timeout) {
440 errorHandler->logError(WARNING, "Reconnection attempt timed out for sensor: " + sensorName);
441 return false;
442 }
443
444 // Special handling for Si7021 sensors based on type string
445 if (sensor->getTypeString().indexOf("Si7021") >= 0) {
446 if (errorHandler) {
447 errorHandler->logError(INFO, "Using specialized reconnection for Si7021 sensor");
448 }
449
450 // First try regular initialize with timeout check
451 if (millis() - startTime < timeout - 100) {
452 if (sensor->initialize()) {
453 if (errorHandler) {
454 errorHandler->logError(INFO, "Successfully reconnected Si7021 sensor via initialize: " + sensorName);
455 }
456 return true;
457 }
458 }
459
460 // Check timeout before attempting self-test
461 if (millis() - startTime >= timeout) {
462 errorHandler->logError(WARNING, "Reconnection initialization timed out for sensor: " + sensorName);
463 return false;
464 }
465
466 // If that didn't work, try a more aggressive approach with self-test
467 if (sensor->performSelfTest()) {
468 if (errorHandler) {
469 errorHandler->logError(INFO, "Successfully reconnected Si7021 sensor via self-test: " + sensorName);
470 }
471 return true;
472 }
473
474 // Still failed - report and return
475 if (errorHandler) {
476 errorHandler->logError(ERROR, "Failed to reconnect Si7021 sensor: " + sensorName);
477 }
478 return false;
479 }
480
481 // Generic reconnection attempt for other sensors with timeout check
482 if (millis() - startTime < timeout - 200) {
483 if (sensor->initialize()) {
484 if (errorHandler) {
485 errorHandler->logError(INFO, "Successfully reconnected sensor: " + sensorName);
486 }
487 return true;
488 }
489 }
490
491 // Check timeout before attempting self-test
492 if (millis() - startTime >= timeout) {
493 errorHandler->logError(WARNING, "Reconnection initialization timed out for sensor: " + sensorName);
494 return false;
495 }
496
497 // Try self-test if initialize didn't work
498 if (sensor->performSelfTest()) {
499 if (errorHandler) {
500 errorHandler->logError(INFO, "Sensor reconnected via self-test: " + sensorName);
501 }
502 return true;
503 }
504
505 if (errorHandler) {
506 errorHandler->logError(ERROR, "Failed to reconnect sensor: " + sensorName);
507 }
508 return false;
509}
