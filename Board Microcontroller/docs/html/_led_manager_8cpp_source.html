137

139
140 // Don't do any other updates if we're in fatal error mode
141 if (fatalErrorActive) {
142 // Just ensure the LED stays red
143 setColor(COLOR_RED, FULL_BRIGHTNESS);
144 return;
145 }
146
147 // Handle identification mode
148 if (identifying) {
149 unsigned long elapsed = currentTime - identifyStartTime;
150 int flashPhase = (elapsed / 250) % 2; // 2Hz = 500ms period, 250ms per state
151 int flashCount = elapsed / 500; // Complete cycles
152
153 if (flashCount >= 10) {
154 // Identification complete after 10 flashes
155 identifying = false;
156 setNormalMode(); // Return to normal mode
157 } else {
158 // Set the LED based on flash phase
159 if (flashPhase == 0) {
160 setColor(COLOR_BLUE, FULL_BRIGHTNESS);
161 } else {
162 setColor(COLOR_OFF, 0);
163 }
164 }
165
166 return; // Exit early - identification takes precedence (except for fatal errors)
167 }
168
169 // Handle error/warning indication timeout
170 if (errorIndicationActive) {

172
173 if (errorElapsed >= WARNING_ERROR_DURATION) {
174 // Error indication complete, return to normal mode
175 errorIndicationActive = false;
176 setNormalMode();
177 } else {
178 // Ensure the error color is still showing
179 setColor(errorIndicationColor, FULL_BRIGHTNESS);
180 }
181
182 return; // Exit early - error indication takes precedence over reading pulse
183 }

188
