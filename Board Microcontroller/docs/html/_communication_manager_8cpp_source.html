174void CommunicationManager::parseCommand(const String& rawCommand, String& command, std::vector<String>& params) {
175 params.clear();
176
177 // Extract command (up to first space)
178 int spacePos = rawCommand.indexOf(' ');
179 if (spacePos > 0) {
180 command = rawCommand.substring(0, spacePos);
181
182 // Extract space-separated parameters
183 String paramsStr = rawCommand.substring(spacePos + 1);
184 paramsStr.trim();
185
186 // Split parameters
187 while (paramsStr.length() > 0) {
188 int nextSpace = paramsStr.indexOf(' ');
189 if (nextSpace > 0) {
190 params.push_back(paramsStr.substring(0, nextSpace));
191 paramsStr = paramsStr.substring(nextSpace + 1);
192 paramsStr.trim();
193 } else {
194 // Last parameter
195 params.push_back(paramsStr);
196 break;
197 }
198 }
199 } else {
200 command = rawCommand;
201 }

212void CommunicationManager::processCommandLine() {
213 // Increased timeout and buffer size
214 constexpr unsigned long COMMAND_TIMEOUT_MS = Constants::Communication::COMMAND_TIMEOUT_MS;
215 constexpr size_t MAX_BUFFER_SIZE = Constants::Communication::MAX_BUFFER_SIZE;
216
217 // Read a complete command with timeout
218 String rawCommand = "";
219 unsigned long startTime = millis();
220
221 // Read until newline, timeout, or buffer full
222 while ((millis() - startTime < COMMAND_TIMEOUT_MS) && (rawCommand.length() < MAX_BUFFER_SIZE)) {
223 if (Serial.available()) {
224 char c = Serial.read();
225 if (c == '\n' || c == '\r') {
226 if (rawCommand.length() > 0) {
227 break; // Complete command found
228 }
229 // Else ignore empty lines
230 } else {
231 rawCommand += c;
232 }
233 }
234 yield(); // Allow other tasks to run
235 }
236
237 // Check if we hit the buffer limit
238 if (rawCommand.length() >= MAX_BUFFER_SIZE) {
239 errorHandler->logError(ERROR, "Command exceeds buffer size limit of " + String(MAX_BUFFER_SIZE) + " characters");
240 }
241
242 if (rawCommand.length() == 0) {
243 return; // No command to process
244 }
245
246 // Trim whitespace and process command
247 rawCommand.trim();
248 errorHandler->logError(INFO, "Processing command: '" +
249 (rawCommand.length() > 50 ?
250 rawCommand.substring(0, 50) + "..." :
251 rawCommand) +
252 "' (" + String(rawCommand.length()) + " bytes)");
253
254 // Parse and process command
255 String command;
256 std::vector<String> params;
257 parseCommand(rawCommand, command, params);
258
259 // Track if the command was recognized by either system
260 bool commandRecognized = false;
261
262 // Try to handle with our command processors
263 if (processCommand(command, params)) {
264 commandRecognized = true;
265 } else {
266 // Before falling back to SCPI parser, check if it's a HELP request
267 if (command.equalsIgnoreCase("HELP") || command.equalsIgnoreCase("?")) {
268 // Provide basic help information
269 Serial.println("Available commands:");
270 Serial.println("*IDN? - Get device identification");
271 Serial.println("MEAS? - Get measurements from all peripherals");
272 Serial.println("MEAS? <sensor>[:measurement] - Get specific measurements");
273 Serial.println("SYST:SENS:LIST? - List all available peripherals");
274 Serial.println("SYST:CONF? - Get device configuration");
275 Serial.println("RESET - Reset the device");
276 Serial.println();
277 commandRecognized = true;
278 } else {
279 // Try SCPI parser as a fallback
280 // Since we cannot easily detect if the SCPI parser recognized the command,
281 // we'll capture the original Serial buffer position to see if anything was output
282 size_t beforeSerialPos = Serial.availableForWrite();
283
284 char buff[rawCommand.length() + 1];
285 rawCommand.toCharArray(buff, rawCommand.length() + 1);
286 scpiParser->ProcessInput(Serial, buff);
287
288 // Check if SCPI parser generated any output
289 size_t afterSerialPos = Serial.availableForWrite();
290 if (afterSerialPos != beforeSerialPos) {
291 commandRecognized = true;
292 }
293 }
294 }
295
296 // If the command wasn't recognized by either system, log a warning and send an error response
297 if (!commandRecognized) {
298 errorHandler->logError(ERROR, "Unrecognized command: '" +
299 (command.length() > 50 ?
300 command.substring(0, 50) + "..." :
301 command) + "'");
302 }
303
304 // Ensure all responses are sent
305 Serial.flush();
306}

318bool CommunicationManager::handleMeasure(const std::vector<String>& params) {
319 std::vector<String> values;
320
321 try {
322 if (params.empty()) {
323 // No sensors specified, use all available
324 auto registry = sensorManager->getRegistry();
325 auto allSensors = registry.getAllSensors();
326
327 errorHandler->logError(INFO, "MEAS: Collecting data from all " + String(allSensors.size()) + " available peripherals");
328
329 for (auto sensor : allSensors) {
330 collectSensorReadings(sensor->getName(), "", values);
331 }
332 } else {
333 // Process parameters efficiently by batching lookups
334 std::map<String, String> sensorRequests;
335
336 // First pass: parse parameters and group by sensor name
337 for (const auto& param : params) {
338 // Check if parameter contains a colon (sensor:measurements format)
339 int colonPos = param.indexOf(':');
340 String sensorName, measurements;
341
342 if (colonPos > 0) {
343 sensorName = param.substring(0, colonPos);
344 measurements = param.substring(colonPos + 1);
345 errorHandler->logError(INFO, "MEAS: Reading " + sensorName + " with measurements: " + measurements);
346 } else {
347 sensorName = param;
348 measurements = "";
349 errorHandler->logError(INFO, "MEAS: Reading " + sensorName + " with all available measurements");
350 }
351
352 // Store most specific measurement request for each sensor
353 if (sensorRequests.find(sensorName) == sensorRequests.end() ||
354 measurements.length() > sensorRequests[sensorName].length()) {
355 sensorRequests[sensorName] = measurements;
356 }
357 }
358
359 // Second pass: collect data for each unique sensor
360 for (const auto& [sensorName, measurements] : sensorRequests) {
361 collectSensorReadings(sensorName, measurements, values);
362 }
363 }
364
365 // Output a single CSV line with all collected values
366 if (!values.empty()) {
367 String csvLine = values[0];
368 for (size_t i = 1; i < values.size(); i++) {
369 csvLine += "," + values[i];
370 }
371 Serial.println(csvLine);
372 Serial.flush(); // Ensure the response is sent immediately
373 errorHandler->logError(INFO, "MEAS: CSV response sent with " + String(values.size()) + " values");
374 } else {
375 errorHandler->logError(WARNING, "MEAS: No measurement values were collected!");
376 Serial.println("ERROR");
377 Serial.flush();
378 }
379
380 } catch (...) {
381 // Catch any other unforeseen errors
382 errorHandler->logError(WARNING, "MEAS: Unexpected exception during measurement");
383 Serial.println("ERROR");
384 Serial.flush();
385 return false;
386 }
387
388 return true;
389}

391void CommunicationManager::collectSensorReadings(const String& sensorName, const String& measurements, std::vector<String>& values) {
392 // Constants for retry logic
393 const int MAX_RETRIES = Constants::Communication::MAX_READING_RETRIES;
394 const int RETRY_DELAY_MS = Constants::Communication::READING_RETRY_DELAY_MS;
395
396 // Find the sensor
397 ISensor* sensor = sensorManager->findSensor(sensorName);
398 bool sensorOk = (sensor != nullptr && sensor->isConnected());
399
400 if (!sensorOk) {
401 errorHandler->logError(WARNING, "Peripheral " + sensorName + " not found or not connected");
402 return;
403 }
404
405 // Determine which measurements to take
406 bool useAllMeasurements = measurements.length() == 0;
407 String upperMeasurements = measurements;
408 upperMeasurements.toUpperCase();
409
410 bool readTemp = useAllMeasurements || upperMeasurements.indexOf("TEMP") >= 0;
411 bool readHum = useAllMeasurements || upperMeasurements.indexOf("HUM") >= 0;
412
413 // Read temperature if requested and supported
414 if (readTemp && sensorOk && sensor->supportsInterface(InterfaceType::TEMPERATURE)) {
415 // Try first reading without any delay
416 TemperatureReading tempReading = sensorManager->getTemperatureSafe(sensorName);
417
418 if (tempReading.valid) {
419 // Fast path - reading succeeded on first try (most common case)
420 values.push_back(String(tempReading.value));
421 } else {
422 // Only if first read failed, apply retry logic with delays
423 bool success = false;
424 String tempValue = "ERROR";
425
426 for (int attempt = 1; attempt < MAX_RETRIES && !success; attempt++) {
427 // Log retry information at INFO level
428 errorHandler->logError(INFO, "Retry #" + String(attempt) + " for temperature reading from " + sensorName);
429 delay(RETRY_DELAY_MS); // Only delay during retries when needed
430
431 try {
432 tempReading = sensorManager->getTemperatureSafe(sensorName);
433
434 if (tempReading.valid) {
435 tempValue = String(tempReading.value);
436 success = true;
437 errorHandler->logError(INFO, "Successfully read temperature from " + sensorName + " after " + String(attempt+1) + " attempts");
438 break;
439 }
440 } catch (...) {
441 // Continue to next retry attempt
442 }
443 }
444
445 values.push_back(success ? tempValue : "ERROR");
446 }
447 }
448
449 // Read humidity if requested and supported
450 if (readHum && sensorOk && sensor->supportsInterface(InterfaceType::HUMIDITY)) {
451 // Try first reading without any delay
452 HumidityReading humReading = sensorManager->getHumiditySafe(sensorName);
453
454 if (humReading.valid) {
455 // Fast path - reading succeeded on first try (most common case)
456 values.push_back(String(humReading.value));
457 } else {
458 // Only if first read failed, apply retry logic with delays
459 bool success = false;
460 String humValue = "ERROR";
461
462 for (int attempt = 1; attempt < MAX_RETRIES && !success; attempt++) {
463 // Log retry information at INFO level
464 errorHandler->logError(INFO, "Retry #" + String(attempt) + " for humidity reading from " + sensorName);
465 delay(RETRY_DELAY_MS); // Only delay during retries when needed
466
467 try {
468 humReading = sensorManager->getHumiditySafe(sensorName);
469
470 if (humReading.valid) {
471 humValue = String(humReading.value);
472 success = true;
473 errorHandler->logError(INFO, "Successfully read humidity from " + sensorName + " after " + String(attempt+1) + " attempts");
474 break;
475 }
476 } catch (...) {
477 // Continue to next retry attempt
478 }
479 }
480
481 values.push_back(success ? humValue : "ERROR");
482 }
483 }

647bool CommunicationManager::handleLogRouting(const std::vector<String>& params) {
648 if (params.empty()) {
649 errorHandler->logError(ERROR, "Format is SYST:LOG <destination>,<severity>");
650 return false;
651 }
652
653 // Check if the parameters are in a single comma-separated string or in two separate parameters
654 String destination, severityStr;
655
656 if (params.size() == 1 && params[0].indexOf(',') > 0) {
657 // Format: "USB,INFO" (comma-separated in a single parameter)
658 int commaPos = params[0].indexOf(',');
659 destination = params[0].substring(0, commaPos);
660 severityStr = params[0].substring(commaPos + 1);
661 } else if (params.size() >= 2) {
662 // Format: "USB" "INFO" (two separate parameters)
663 destination = params[0];
664 severityStr = params[1];
665 } else {
666 errorHandler->logError(ERROR, "Format is SYST:LOG <destination>,<severity>");
667 return false;
668 }
669
670 destination.toUpperCase();
671 severityStr.toUpperCase();
672
673 // Determine the target stream based on destination
674 Print* targetStream = nullptr;
675 if (destination == "USB" || destination == "SERIAL") {
676 targetStream = &Serial;
677 } else if (destination == "UART" || destination == "DEBUG") {
678 targetStream = uartDebugSerial;
679 } else if (destination == "NONE" || destination == "OFF") {
680 targetStream = nullptr;
681 } else {
682 errorHandler->logError(ERROR, "Invalid destination. Use USB, UART, or NONE");
683 return false;
684 }
685
686 // Determine severity
687 ErrorSeverity minSeverity = ErrorHandler::stringToSeverity(severityStr);
688
689 // Enable custom routing if it's not already enabled
690 errorHandler->enableCustomRouting(true);
691
692 // Set the routing configuration
693 errorHandler->setOutputSeverity(targetStream, minSeverity);
694
695 errorHandler->logError(INFO, "Log routing updated: " + destination + " will show " +
696 ErrorHandler::severityToString(minSeverity) + " and higher");
697
698 return true;
699}

718 }
719 // Log the message
720 bool isFatal = errorHandler->logError(severity, message);
721
722 // Extra direct LED control for debugging
723 if (ledManager && severity >= WARNING) {
724 if (severity == WARNING) ledManager->indicateWarning();
725 if (severity == ERROR) ledManager->indicateError();
726 if (severity == FATAL) ledManager->indicateFatalError();
727 }
728
729 // For FATAL errors, handle specially
730 if (isFatal) {
731 // For FATAL, we might want to reset after some delay
732 int resetDelay = -1;
733 if (params.size() > 1) {
734 resetDelay = params[1].toInt();
735 }
736
737 if (resetDelay > 0) {
738 errorHandler->logError(FATAL, "Device will reset after " + String(resetDelay) + "ms");
739 Serial.flush();
740 delay(resetDelay);
741 ESP.restart();
742 } else {
743 errorHandler->logError(FATAL, "Fatal error - device halted");
744 Serial.flush();
745 // Enter infinite loop, but keep LED updated
746 while (true) {
747 if (ledManager) {
748 ledManager->update();
749 }
750 delay(100);
751 yield();
752 }
753 }
754 }
755
756 return true;
