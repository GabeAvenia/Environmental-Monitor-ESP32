23bool PT100Sensor::initialize() {
24 errorHandler->logError(INFO, "Initializing PT100 RTD sensor: " + name + " on SS pin " + String(ssPin));
25
26 // Make sure SPI is initialized
27 if (!spiManager || !spiManager->isInitialized()) {
28 errorHandler->logError(ERROR, "SPI not initialized for PT100 sensor: " + name);
29 connected = false;
30 return false;

32
33 // Register the SS pin with the SPI manager
34 spiManager->registerSSPin(ssPin);
35
36 // Begin the MAX31865 with the appropriate wiring config
37 max31865_numwires_t wiringMode;
38 switch (numWires) {
39 case 2:
40 wiringMode = MAX31865_2WIRE;
41 break;
42 case 3:
43 wiringMode = MAX31865_3WIRE;
44 break;
45 case 4:
46 default:
47 wiringMode = MAX31865_4WIRE;
48 break;
49 }
50
51 // Initialize the MAX31865
52 max31865.begin(wiringMode);
53
54 // Check for any faults
55 uint8_t fault = max31865.readFault();
56 if (fault) {
57 errorHandler->logError(ERROR, "MAX31865 fault detected during initialization: " + getFaultStatus());
58 max31865.clearFault();
59 // Only fail initialization for critical faults
60 if (fault & (MAX31865_FAULT_OVUV | MAX31865_FAULT_REFINHIGH | MAX31865_FAULT_REFINLOW)) {
61 connected = false;
62 return false;
63 }
64 errorHandler->logError(ERROR, "Non-critical fault detected, attempting to continue");
65 }
66
67 // Read the RTD value directly for diagnostics
68 uint16_t rtd = max31865.readRTD();
69 float ratio = rtd / 32768.0;
70 float resistance = ratio * rRef;
71
72 errorHandler->logError(INFO, "Initial PT100 RTD value: " + String(rtd));
73 errorHandler->logError(INFO, "Initial PT100 resistance: " + String(resistance) + " ohms (ratio: " + String(ratio, 8) + ")");
74
75 // Check if RTD value is zero, which indicates a likely connection issue
76 if (rtd == 0) {
77 errorHandler->logError(ERROR, "RTD value is 0, suggesting a connection problem. Check wiring and SPI communication.");
78 // Let's not mark it as disconnected yet, to allow for diagnostics
79 }
80
81 // Get initial temperature reading
82 float temp = max31865.temperature(PT100_RTD_VALUE, rRef);
83 errorHandler->logError(INFO, "Initial PT100 temperature: " + String(temp) + "°C");
84
85 // Still mark as connected even with suspicious values for diagnostic purposes

88 // Store the initial reading
