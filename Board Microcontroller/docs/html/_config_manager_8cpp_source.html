443// Update configuration from JSON string
444bool ConfigManager::updateConfigFromJson(const String& jsonConfig) {
445 // Log a shorter version of the config to avoid huge logs
446 errorHandler->logError(INFO, "Received config update: " +
447 jsonConfig.substring(0, std::min(50, (int)jsonConfig.length())) +
448 (jsonConfig.length() > 50 ? "..." : ""));
449
450 // Make a backup of the current configuration
451 String backupConfig = getConfigJson();
452
453 // Extract clean JSON data by finding the opening brace
454 int jsonStart = jsonConfig.indexOf('{');
455 if (jsonStart < 0) {
456 errorHandler->logError(ERROR, "No JSON object found in config");
457 return false;
458 }
459 String cleanJson = jsonConfig.substring(jsonStart);
460
461 // Parse the JSON
462 JsonDocument doc;
463 DeserializationError error = deserializeJson(doc, cleanJson);
464
465 if (error || doc.overflowed()) {
466 errorHandler->logError(ERROR, "Failed to parse JSON config: " +
467 (error ? String(error.c_str()) : "Document too large"));
468 return false;
469 }
470
471 // Save to file
472 if (!writeConfigToFile(doc)) {
473 errorHandler->logError(ERROR, "Failed to write new configuration to file");
474 return false;
475 }
476
477 // Reload and notify
478 bool success = loadConfigFromFile();
479 if (success) {
480 errorHandler->logError(INFO, "Config reloaded successfully, notifying listeners");
481 notifyConfigChanged(jsonConfig);
482 } else {
483 errorHandler->logError(ERROR, "Failed to reload configuration, rolling back to previous state");
484
485 // Restore the previous configuration
486 JsonDocument backupDoc;
487 DeserializationError backupError = deserializeJson(backupDoc, backupConfig);
488
489 if (!backupError && writeConfigToFile(backupDoc)) {
490 errorHandler->logError(INFO, "Successfully rolled back to previous configuration");
491 loadConfigFromFile(); // Reload the backup configuration
492 } else {
493 errorHandler->logError(ERROR, "Failed to roll back to previous configuration");
494 }
495 }
496
497 return success;
498}

706// Update only the additional configuration
707bool ConfigManager::updateAdditionalConfigFromJson(const String& jsonConfig) {
708 // Handle empty input - erase additional config with warning
709 if (jsonConfig.length() == 0 || jsonConfig == "{}" || jsonConfig == "null") {
710 errorHandler->logError(WARNING, "Empty additional configuration received - clearing additional section");
711 additionalConfig = "";
712
713 // Update the configuration file
714 JsonDocument doc;
715 if (!readConfigFromFile(doc)) {
716 errorHandler->logError(ERROR, "Failed to read existing configuration");
717 return false;
718 }
719
720 // Remove Additional field if it exists
721 if (doc["Additional"].is<JsonObject>()) {
722 doc.remove("Additional");
723 }
724
725 // Write updated configuration to file
726 return writeConfigToFile(doc);
727 }
728
729 // Clean the input JSON and parse it
730 String cleanJson = jsonConfig;
731 int jsonStart = cleanJson.indexOf('{');
732 if (jsonStart > 0) {
733 cleanJson = cleanJson.substring(jsonStart);
734 }
735
736 JsonDocument doc;
737 DeserializationError error = deserializeJson(doc, cleanJson);
738
739 if (error || doc.overflowed()) {
740 errorHandler->logError(ERROR, "Failed to parse additional configuration JSON: " +
741 (error ? String(error.c_str()) : "Document too large"));
742 return false;
743 }
744
745 // Extract Additional configuration if present
746 String newAdditionalConfig = "";
747 if (doc["Additional"].is<JsonObject>()) {
748 // Serialize the Additional field to a string if it's an object
749 serializeJson(doc["Additional"], newAdditionalConfig);
750 } else if (doc["Additional"].is<String>()) {
751 // Use the string directly if it's a string
752 newAdditionalConfig = doc["Additional"].as<String>();
753 } else {
754 // Convert to string if it's another type
755 newAdditionalConfig = doc["Additional"].as<String>();
756 }
757
758 // Backup existing additional config
759 String oldAdditionalConfig = additionalConfig;
760
761 // Update additional config
762 additionalConfig = newAdditionalConfig;
763
764 // Read current configuration file to get other parts
765 JsonDocument fullDoc;
766 if (!readConfigFromFile(fullDoc)) {
767 // Failed to read existing config, revert to old additional config
768 additionalConfig = oldAdditionalConfig;
769 return false;
770 }
771
772 // Update the Additional field
773 if (additionalConfig.length() > 0) {
774 // Parse the additionalConfig to a JSON object
775 JsonDocument additionalDoc;
776 DeserializationError additionalError = deserializeJson(additionalDoc, additionalConfig);
777
778 if (!additionalError && !additionalConfig.startsWith("\"")) {
779 // If it's a valid JSON object and not a JSON string, store as object
780 fullDoc["Additional"] = additionalDoc;
781 } else {
782 // If not a valid JSON object or it's a JSON string, store as raw string
783 fullDoc["Additional"] = additionalConfig;
784 }
785 }
786 // Write updated configuration to file
787 if (!writeConfigToFile(fullDoc)) {
788 // If writing fails, revert to old configuration
789 additionalConfig = oldAdditionalConfig;
790 errorHandler->logError(ERROR, "Failed to write updated additional configuration");
791 return false;
792 }
793
794 errorHandler->logError(INFO, "Additional configuration updated successfully");
795
796 // Notify about the configuration change
797 String configJson;
798 serializeJson(fullDoc, configJson);
799 notifyConfigChanged(configJson);
800
801 return true;
802}
