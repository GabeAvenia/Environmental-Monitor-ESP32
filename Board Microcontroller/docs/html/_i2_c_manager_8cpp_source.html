117 return false;

120 errorHandler->logError(INFO, "Found " + String(addresses.size()) + " I2C devices on port " +
121 portToString(port));

156String I2CManager::portToString(I2CPort port) {
157 int portValue = static_cast<int>(port);
158
159 // Handle built-in ports
160 switch (port) {
161 case I2CPort::I2C0: return "I2C0";
162 case I2CPort::I2C1: return "I2C1";
163
164 default: break;
165 }
166
167 // Handle multiplexed ports
168 if (portValue >= static_cast<int>(I2CPort::I2C_MULTIPLEXED_START)) {
169 int channel = portValue - static_cast<int>(I2CPort::I2C_MULTIPLEXED_START);
170 return "I2C_MUX_" + String(channel);
171 }

173 return "UNKNOWN";
