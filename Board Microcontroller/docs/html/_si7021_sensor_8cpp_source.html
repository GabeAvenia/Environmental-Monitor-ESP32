43bool Si7021Sensor::updateReadings() const {
44 if (!connected) {
45 errorHandler->logError(ERROR, "Attempted to read from disconnected sensor: " + name);
46 return false;

48
49 // Adding retries for more robustness
50 const int MAX_RETRIES = 3;
51 float temp = NAN;
52 float humidity = NAN;
53 bool success = false;
54
55 for (int attempt = 0; attempt < MAX_RETRIES && !success; attempt++) {
56 if (attempt > 0) {
57 delay(50);
58 errorHandler->logError(INFO, "Retrying Si7021 reading, attempt " + String(attempt+1) + " of " + String(MAX_RETRIES));
59 }
60

63
64 if (!isnan(temp)) {

67
68 if (!isnan(humidity)) {
69 success = true;
70 }
71 }

74 delay(20);
75 }
76 }
77
78 if (!success) {
79 errorHandler->logError(ERROR, "Failed to read from Si7021 sensor: " + name + " after " + String(MAX_RETRIES) + " retries");
80 const_cast<Si7021Sensor*>(this)->connected = false;
81 return false;
82 }
83

85 lastHumidity = humidity;
